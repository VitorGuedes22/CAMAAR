def warn_error?(num, marker)
  klass = ""

  if @filter.error?(num)
    klass = ' class="error"'
    @errors<< [@current, marker, num]
  elsif @filter.warn?(num)
    klass = ' class="warning"'
    @warnings<< [@current, marker, num]
  end

  klass
end
